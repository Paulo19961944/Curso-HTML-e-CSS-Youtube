             // Chama os Elementos HTML
            const display = document.querySelector(".display");
            const numberButtons = document.querySelectorAll(".number");
            const operatorButtons = document.querySelectorAll(".operator");
            const equalButton = document.querySelector(".equal");
            const cleanNumber = document.querySelector(".clean-number")
            const cleanAll = document.querySelector(".clean-all")

            // Variáveis
            let firstNumber = null;
            let secondNumber = null;
            let currentOperation = null;

            // Adiciona eventos de clique aos botões de número
            numberButtons.forEach(button => {
                button.addEventListener("click", () => {
                    display.textContent += button.textContent;
                });
            });

            // Adiciona eventos de clique aos botões de operação
            operatorButtons.forEach(button => {
                button.addEventListener("click", () => {
                    if (firstNumber === null) {
                        firstNumber = parseFloat(display.textContent);
                        currentOperation = button.textContent;
                        display.textContent = "";
                    } else if (secondNumber === null) {
                        secondNumber = parseFloat(display.textContent);
                        calculateResult();
                        currentOperation = button.textContent;
                        display.textContent = "";
                    }
                });
            });

            // Adiciona evento de clique ao botão de igual
            equalButton.addEventListener("click", () => {
                if (firstNumber !== null && secondNumber === null) {
                    secondNumber = parseFloat(display.textContent);
                    calculateResult();
                }
            });

            cleanNumber.addEventListener("click", () => {
                if (currentOperation !== null) {
                    // Se uma operação já foi selecionada e o segundo número foi digitado, limpa apenas o segundo número
                    const displayText = display.textContent;
                    const lastOperatorIndex = displayText.lastIndexOf(currentOperation);
                    display.textContent = displayText.substring(0, lastOperatorIndex + 1);
                    secondNumber = null; // Reseta o segundo número
                } else {
                    // Se nenhum segundo número foi digitado, limpa o display inteiro
                    cleanAll();
                }
            });

            cleanAll.addEventListener("click", () => {
                display.textContent = "";
                firstNumber = null;
                secondNumber = null;
                currentOperation = null;
            });

            // Função para calcular o resultado da operação
            function calculateResult() {
                let result = null;
                switch (currentOperation) {
                    case "+":
                        result = firstNumber + secondNumber;
                        break;
                    case "-":
                        result = firstNumber - secondNumber;
                        break;
                    case "*":
                        result = firstNumber * secondNumber;
                        break;
                    case "/":
                        result = firstNumber / secondNumber;
                        break;
                }

                // Reseta tudo
                display.textContent = result.toFixed(2); // Exibe o Resultado com 2 casas decimais
                firstNumber = result;
                secondNumber = null;
            }
